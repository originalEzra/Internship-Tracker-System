# 12.1 Get Status History

```http
GET {{baseUrl}}/api/internships/{{internshipId}}/status-history
Authorization: Bearer {{token}}
```

Assertions:

```javascript
pm.test("status history records successful transition", function () {
  pm.response.to.have.status(200);
  const json = pm.response.json();
  pm.expect(json.code).to.eql(200);
  pm.expect(json.data).to.be.an("array");
  pm.expect(json.data.length).to.be.greaterThan(0);
  const history = json.data[0];
  pm.expect(history.internshipId).to.eql(Number(pm.collectionVariables.get("internshipId")));
  pm.expect(history.fromStatus).to.eql("APPLIED");
  pm.expect(history.toStatus).to.eql("ONLINE_ASSESSMENT");
  pm.expect(history.note).to.eql("Received OA");
  pm.expect(history.operatorUserId).to.eql(Number(pm.collectionVariables.get("userId")));
  pm.expect(history.createdAt).to.be.a("string");
});
```
